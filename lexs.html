About Me

Hello! I'm Lex, a 14-year-old photographer with a passion for capturing nature in its rawest form.

I find magic in the way light dances through tree branches, how water reflects the world around it, and those quiet moments where nature reveals its true beauty. My camera is my way of preserving these fleeting instances that might otherwise go unnoticed.

Rather than focusing on portraits or people, I'm drawn to the simplicity and authenticity of natural landscapes. Each photograph I take is a moment frozen in time - the perfect ray of sunlight, a drop of rain suspended in mid-air, or the intricate patterns in leaves and branches.

What makes my work unique is how I enhance these natural moments with subtle animations and effects. Imagine seeing gentle rain falling across a forest scene, or the warm glow of firelight illuminating a twilight landscape. These digital enhancements don't replace the natural beauty I've captured - they amplify it, creating an immersive experience that brings you closer to how that moment felt in person.

Through my photography, I hope to remind people of the quiet joy that comes from reconnecting with nature. In our busy lives, we often forget to pause and notice the extraordinary beauty in ordinary places. My goal is to help you see the world through fresh eyes, rediscovering the wonder that surrounds us every day.

"Photography is about capturing nature's poetry through light and time."

// Parameters const gravity = 0.4; const bounce = 0.8; let velocity = 0; const animate = () => { ctx.clearRect(0, 0, width, height); // Apply gravity velocity += gravity; y += velocity; // Check for floor collision if (y + size > height) { y = height - size; velocity = -velocity * bounce; } // Draw ball ctx.beginPath(); ctx.arc(centerX, y, size, 0, Math.PI * 2); ctx.fillStyle = '#4285F4'; ctx.fill(); ctx.strokeStyle = '#3367D6'; ctx.lineWidth = 3; ctx.stroke(); animationRef.current = requestAnimationFrame(animate); }; animate(); } function animateShake(canvas, ctx) { const width = canvas.width; const height = canvas.height; const size = 80; const centerX = width / 2; }; animate(); } function animatePulse(canvas, ctx) { const width = canvas.width; const height = canvas.height; const centerX = width / 2; const centerY = height / 2; const baseSize = 40; let time = 0; flake.y += flake.speed; // Reset if off screen if (flake.y > height) { flake.y = -10; flake.x = Math.random() * width; } // Draw snowflake ctx.beginPath(); ctx.arc(flake.x, flake.y, flake.size, 0, Math.PI * 2); ctx.fill(); }); animationRef.current = requestAnimationFrame(animate); }; animate(); } function animateSparkles(canvas, ctx) { const width = canvas.width; const height = canvas.height; // Create sparkles const sparkles = []; ctx.fill(); // Add glow ctx.beginPath(); ctx.arc(sparkle.x, sparkle.y, sparkle.size * sparkle.life * 2, 0, Math.PI * 2); ctx.fillStyle = `hsla(${sparkle.hue}, 100%, 50%, ${sparkle.life * 0.3})`; ctx.fill(); }); animationRef.current = requestAnimationFrame(animate); }; animate(); } // --------- ADVANCED ANIMATIONS --------- function animateGalaxy(canvas, ctx) { const width = canvas.width; const height = canvas.height; const centerX = width / 2; const centerY = height / 2; // Create stars const stars = []; const starCount = 600; for (let i = 0; i < starCount; i++) { const angle = Math.random() * Math.PI * 2; const distance = Math.random() * Math.random() * 150; // Quadratic distribution for density const armOffset = Math.sin(angle * 4) * 0.3; // Create spiral arms stars.push({ x: 0, y: 0, size: Math.random() * 2 + 0.5, angle: angle + armOffset, distance: distance, orbitSpeed: 0.002 / (Math.sqrt(distance) * 0.1 + 0.1), hue: (angle / Math.PI / 2) * 60 + 220, brightness: 50 + Math.random() * 30 }); } let rotation = 0; const animate = () => { // Clear with fade for trails ctx.fillStyle = 'rgba(0, 0, 5, 0.15)'; ctx.fillRect(0, 0, width, height); // Update rotation rotation += 0.002; // Draw black hole at center const blackHoleGradient = ctx.createRadialGradient(centerX, centerY, 0, centerX, centerY, 20); blackHoleGradient.addColorStop(0, 'rgba(30, 0, 60, 0.5)'); blackHoleGradient.addColorStop(1, 'rgba(0, 0, 20, 0)'); ctx.beginPath(); ctx.arc(centerX, centerY, 20, 0, Math.PI * 2); ctx.fillStyle = blackHoleGradient; ctx.fill(); // Draw stars stars.forEach(star => { // Update position star.angle += star.orbitSpeed; star.x = centerX + Math.cos(star.angle) * star.distance; star.y = centerY + Math.sin(star.angle) * star.distance; // Draw star ctx.beginPath(); ctx.arc(star.x, star.y, star.size, 0, Math.PI * 2); ctx.fillStyle = `hsl(${star.hue}, 80%, ${star.brightness}%)`; ctx.fill(); }); animationRef.current = requestAnimationFrame(animate); }; animate(); } function animateFire(canvas, ctx) { const width = canvas.width; const height = canvas.height; // Fire particles const particles = []; const particleCount = 150; for (let i = 0; i < particleCount; i++) { particles.push({ x: width / 2 + (Math.random() - 0.5) * 60, y: height - 20, size: Math.random() * 5 + 2, speed: Math.random() * 3 + 2, life: Math.random() * 0.5 + 0.5, hue: Math.random() * 50 + 10 }); } const animate = () => { // Clear with fade ctx.fillStyle = 'rgba(0, 0, 0, 0.1)'; ctx.fillRect(0, 0, width, height); // Draw fire base const baseGradient = ctx.createRadialGradient(width/2, height, 0, width/2, height, 60); baseGradient.addColorStop(0, 'rgba(255, 100, 20, 0.8)'); baseGradient.addColorStop(1, 'rgba(255, 100, 20, 0)'); ctx.beginPath(); ctx.arc(width/2, height, 60, 0, Math.PI, true); ctx.fillStyle = baseGradient; ctx.fill(); // Update and draw particles particles.forEach(p => { // Update p.y -= p.speed; p.x += (Math.random() - 0.5) * 2; p.life -= 0.01; p.size -= 0.05; // Reset if dead if (p.life <= 0 || p.size <= 0) { p.x = width / 2 + (Math.random() - 0.5) * 60; p.y = height - 20; p.size = Math.random() * 5 + 2; p.life = Math.random() * 0.5 + 0.5; p.hue = Math.random() * 50 + 10; } // Draw particle ctx.beginPath(); ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2); // Color based on life and initial hue const lightness = 50 + 50 * p.life; ctx.fillStyle = `hsla(${p.hue}, 100%, ${lightness}%, ${p.life})`; ctx.fill(); }); animationRef.current = requestAnimationFrame(animate); }; animate(); } function animateLightning(canvas, ctx) { const width = canvas.width; const height = canvas.height; // Lightning bolts const bolts = []; let nextBoltTime = 0; // Generate lightning path recursively const generateLightningPath = (startX, startY, endX, endY, displacementFactor, segments = []) => { if (segments.length === 0) { segments.push({ x: startX, y: startY }); } if (displacementFactor < 2) { segments.push({ x: endX, y: endY }); return segments; } const midX = (startX + endX) / 2; const midY = (startY + endY) / 2; // Add displacement to midpoint const displacement = (Math.random() * 2 - 1) * displacementFactor; const perpX = -(endY - startY); const perpY = endX - startX; // Normalize perpendicular vector const perpLength = Math.sqrt(perpX * perpX + perpY * perpY); if (perpLength > 0) { const midPointX = midX + (perpX / perpLength) * displacement; const midPointY = midY + (perpY / perpLength) * displacement; // Recursively generate path generateLightningPath(startX, startY, midPointX, midPointY, displacementFactor / 2, segments); generateLightningPath(midPointX, midPointY, endX, endY, displacementFactor / 2, segments); } return segments; }; const createLightningBolt = () => { const startX = Math.random() * width; const endX = Math.random() * (width * 0.8) + width * 0.1; const segments = generateLightningPath(startX, 0, endX, height, 80); bolts.push({ segments, life: 1, decay: 0.05 + Math.random() * 0.05 }); // Create a few branches const branchCount = Math.floor(Math.random() * 3) + 1; for (let i = 0; i < branchCount; i++) { const segmentIndex = Math.floor(Math.random() * (segments.length - 2)) + 1; const segment = segments[segmentIndex]; // Branch endpoint const angle = Math.random() * Math.PI - Math.PI / 2; // Mostly downward const length = Math.random() * height / 3 + 30; const endX = segment.x + Math.cos(angle) * length; const endY = segment.y + Math.sin(angle) * length; const branchSegments = generateLightningPath(segment.x, segment.y, endX, endY, 40); bolts.push({ segments: branchSegments, life: 1, decay: 0.07 + Math.random() * 0.05 }); } }; const animate = (timestamp) => { // Dark background with fade ctx.fillStyle = 'rgba(10, 15, 30, 0.15)'; ctx.fillRect(0, 0, width, height); // Create new bolt occasionally if (timestamp > nextBoltTime) { createLightningBolt(); nextBoltTime = timestamp + Math.random() * 2000 + 500; } // Update and draw lightning bolts for (let i = bolts.length - 1; i >= 0; i--) { const bolt = bolts[i]; // Update life bolt.life -= bolt.decay; // Remove dead bolts if (bolt.life <= 0) { bolts.splice(i, 1); continue; } // Draw bolt if (bolt.segments.length > 1) { // Glow effect ctx.beginPath(); ctx.moveTo(bolt.segments[0].x, bolt.segments[0].y); for (let j = 1; j < bolt.segments.length; j++) { ctx.lineTo(bolt.segments[j].x, bolt.segments[j].y); } ctx.strokeStyle = `rgba(120, 170, 255, ${bolt.life * 0.3})`; ctx.lineWidth = 6; ctx.stroke(); // Core ctx.beginPath(); ctx.moveTo(bolt.segments[0].x, bolt.segments[0].y); for (let j = 1; j < bolt.segments.length; j++) { ctx.lineTo(bolt.segments[j].x, bolt.segments[j].y); } ctx.strokeStyle = `rgba(200, 230, 255, ${bolt.life})`; ctx.lineWidth = 2; ctx.stroke(); } } animationRef.current = requestAnimationFrame(animate); }; animate(0); } function animateFlocking(canvas, ctx) { const width = canvas.width; const height = canvas.height; // Boids (birds/fish) const boids = []; const boidCount = 70; // Boid parameters const visualRange = 50; const minDistance = 20; const maxSpeed = 3; const maxForce = 0.05; // Create boids for (let i = 0; i < boidCount; i++) { boids.push({ x: Math.random() * width, y: Math.random() * height, vx: Math.random() * 2 - 1, vy: Math.random() * 2 - 1, color: `hsl(${Math.random() * 60 + 180}, 100%, 50%)` }); } // Separation: avoid crowding neighbors const separation = (boid, boids) => { let vx = 0; let vy = 0; let count = 0; for (let i = 0; i < boids.length; i++) { const other = boids[i]; const dx = boid.x - other.x; const dy = boid.y - other.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance > 0 && distance < minDistance) { vx += dx / distance; vy += dy / distance; count++; } } if (count > 0) { vx /= count; vy /= count; } return [vx, vy]; }; // Alignment: steer towards average heading of neighbors const alignment = (boid, boids) => { let vx = 0; let vy = 0; let count = 0; for (let i = 0; i < boids.length; i++) { const other = boids[i]; const dx = boid.x - other.x; const dy = boid.y - other.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance > 0 && distance < visualRange) { vx += other.vx; vy += other.vy; count++; } } if (count > 0) { vx /= count; vy /= count; // Normalize and scale const length = Math.sqrt(vx * vx + vy * vy); if (length > 0) { vx /= length; vy /= length; } // Subtract current velocity vx -= boid.vx; vy -= boid.vy; } return [vx, vy]; }; // Cohesion: steer towards center of neighbors const cohesion = (boid, boids) => { let cx = 0; let cy = 0; let count = 0; for (let i = 0; i < boids.length; i++) { const other = boids[i]; const dx = boid.x - other.x; const dy = boid.y - other.y; const distance = Math.sqrt(dx * dx + dy * dy); if (distance > 0 && distance < visualRange) { cx += other.x; cy += other.y; count++; } } let vx = 0; let vy = 0; if (count > 0) { cx /= count; cy /= count; // Vector towards center vx = cx - boid.x; vy = cy - boid.y; // Normalize and scale const length = Math.sqrt(vx * vx + vy * vy); if (length > 0) { vx /= length; vy /= length; } } return [vx, vy]; }; const animate = () => { // Clear with fade for trails ctx.fillStyle = 'rgba(0, 20, 40, 0.1)'; ctx.fillRect(0, 0, width, height); // Update boids boids.forEach(boid => { // Apply flocking rules const [vx1, vy1] = separation(boid, boids); const [vx2, vy2] = alignment(boid, boids); const [vx3, vy3] = cohesion(boid, boids); // Update velocity boid.vx += vx1 * 0.1 + vx2 * 0.05 + vx3 * 0.05; boid.vy += vy1 * 0.1 + vy2 * 0.05 + vy3 * 0.05; // Limit speed const speed = Math.sqrt(boid.vx * boid.vx + boid.vy * boid.vy); if (speed > maxSpeed) { boid.vx = (boid.vx / speed) * maxSpeed; boid.vy = (boid.vy / speed) * maxSpeed; } // Update position boid.x += boid.vx; boid.y += boid.vy; // Wrap around edges if (boid.x < 0) boid.x = width; if (boid.y < 0) boid.y = height; if (boid.x > width) boid.x = 0; if (boid.y > height) boid.y = 0; // Draw boid as triangle const angle = Math.atan2(boid.vy, boid.vx); ctx.save(); ctx.translate(boid.x, boid.y); ctx.rotate(angle); ctx.beginPath(); ctx.moveTo(10, 0); ctx.lineTo(-5, 5); ctx.lineTo(-5, -5); ctx.closePath(); ctx.fillStyle = boid.color; ctx.fill(); ctx.restore(); }); animationRef.current = requestAnimationFrame(animate); }; animate(); } function animateAurora(canvas, ctx) { const width = canvas.width; const height = canvas.height; // Aurora waves const waves = []; const waveCount = 5; for (let i = 0; i < waveCount; i++) { waves.push({ points: [], color: `hsla(${120 + i * 30}, 80%, 60%, 0.2)`, speed: 0.003 + i * 0.001, offset: i * 0.2 }); // Generate points for each wave const pointCount = Math.floor(width / 10); for (let j = 0; j < pointCount; j++) { waves[i].points.push({ x: j * (width / (pointCount - 1)), y: 0, noiseOffset: Math.random() * 100 }); } } let time = 0; const animate = () => { // Update time time += 0.01; // Background gradient const gradient = ctx.createLinearGradient(0, 0, 0, height); gradient.addColorStop(0, '#000510'); gradient.addColorStop(1, '#001030'); ctx.fillStyle = gradient; ctx.fillRect(0, 0, width, height); // Draw stars for (let i = 0; i < 100; i++) { const x = i * (width / 100) + Math.sin(time + i * 0.1) * 1; const y = (i % 20) * (height / 20) + Math.cos(time + i * 0.1) * 1; const size = Math.random() * 2 + 0.5; ctx.beginPath(); ctx.arc(x, y, size, 0, Math.PI * 2); ctx.fillStyle = 'rgba(255, 255, 255, 0.8)'; ctx.fill(); } // Draw waves from back to front waves.forEach(wave => { // Update points wave.points.forEach(point => { const noiseValue = Math.sin(point.x * 0.02 + time * wave.speed + wave.offset); const waveHeight = height * 0.5; point.y = height * 0.7 + noiseValue * waveHeight; }); // Draw wave ctx.beginPath(); ctx.moveTo(0, height); ctx.lineTo(wave.points[0].x, wave.points[0].y); for (let i = 1; i < wave.points.length; i++) { const xc = (wave.points[i].x + wave.points[i - 1].x) / 2; const yc = (wave.points[i].y + wave.points[i - 1].y) / 2; ctx.quadraticCurveTo(wave.points[i - 1].x, wave.points[i - 1].y, xc, yc);
